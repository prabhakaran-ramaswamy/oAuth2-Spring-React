# Command to generate keystore.jks for SSL/HTTPS:
keytool -genkeypair -alias jpa-ssl-cert -keyalg RSA -keystore d:\eclipse-w\jpa-ssl\keystore.p12 -storepass changeit -keypass changeit -keysize 2048 -validity 3650 -storetype PKCS12 -dname "CN=localhost, OU=Dev, O=MyOrg, L=Chennai, S=Tamilnadu, C=IN"

keytool -exportcert -alias jpa-ssl-cert -keystore D:\eclipse-w\jpa-ssl\keystore.p12 -storetype PKCS12 -storepass changeit -file D:\eclipse-w\jpa-ssl\server.crt


keytool -importcert -alias server-cert -file D:\eclipse-w\jpa-ssl\server.crt -keystore D:\eclipse-w\jpa-ssl\truststore.p12   -storetype PKCS12 -storepass changeit
